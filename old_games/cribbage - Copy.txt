package cribbage

import (
	"fmt"
	"math/rand"
	"time"
)

// Enumerations (Typed Constants)
type Suit int

const (
	Clubs Suit = iota
	Diamonds
	Hearts
	Spades
)

type Color int

const (
	Black Color = iota
	Red
)

type Rank int

const (
	Ace Rank = iota + 1
	Two
	Three
	Four
	Five
	Six
	Seven
	Eight
	Nine
	Ten
	Jack
	Queen
	King
)

// Card Data Structure
type Card struct {
	Rank  Rank
	Suit  Suit
	Color Color
}

func (c Card) ValueMax10() int {
	switch c.Rank {
	case Jack, Queen, King:
		return 10
	default:
		return int(c.Rank)
	}
}

// Value used for runs
func (c Card) Value() int {
	return int(c.Rank)
}

// String Conversions for CLI
func (r Rank) String() string {
	switch r {
	case Ace:
		return "A"
	case Jack:
		return "J"
	case Queen:
		return "Q"
	case King:
		return "K"
	default:
		return string('0' + rune(r))
	}
}

func (s Suit) String() string {
	switch s {
	case Clubs:
		return "♣"
	case Diamonds:
		return "♦"
	case Hearts:
		return "♥"
	case Spades:
		return "♠"
	default:
		return "?"
	}
}

// Deck of Cards
type Deck []Card

func NewDeck() Deck {
	// len 0 and cap 52
	deck := make(Deck, 0, 52)
	// Clubs 0, Diamonds 1, Hearts 2, Spades 3
	for suit := Clubs; suit <= Spades; suit++ {
		color := Black
		if suit == Diamonds || suit == Hearts {
			color = Red
		}
		for rank := Ace; rank <= King; rank++ {
			deck = append(deck, Card{rank, suit, color})
		}
	}

	return deck
}

func (d Deck) Shuffle() {
	rand.Seed(time.Now().UnixNano())
	// TODO implement my own Fisher-Yates shuffle
	rand.Shuffle(len(d), func(i, j int) {
		d[i], d[j] = d[j], d[i]
	})
}

// Player Hand from Deck
type Hand []Card

func Deal(deck Deck) (Hand, Hand, Deck) {
	count := 6
	p1 := make(Hand, 0, count)
	p2 := make(Hand, 0, count)
	for i := 0; i < count; i++ {
		//for i := range count {
		p1 = append(p1, deck[0])
		p2 = append(p2, deck[1])
		deck = deck[2:]
	}
	return p1, p2, deck
}

func main() {
	var myDeck Deck = NewDeck()
	myDeck.Shuffle()
	hand1, hand2, _ := Deal(myDeck)
